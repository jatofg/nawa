73void File::writeToDisk(std::string const& path) const {

75 ios_base::iostate exceptionMask = outfile.exceptions() | ios::failbit;
76 outfile.exceptions(exceptionMask);

78 outfile.open(path, ofstream::out | ofstream::binary);

80 outfile.close();
