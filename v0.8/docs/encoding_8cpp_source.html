133 string encoding::htmlEncode(string input, bool encodeAll) {
134  if (!encodeAll) {
135  boost::replace_all(input, "&", "&amp;");
136  boost::replace_all(input, "\"", "&quot;");
137  boost::replace_all(input, "<", "&lt;");
138  boost::replace_all(input, ">", "&gt;");
139  } else {
140  // convert to utf32 to iterate through the characters
141  wstring_convert<codecvt_utf8<char32_t>, char32_t> cv;
142  u32string uinput = cv.from_bytes(input);
143  ostringstream output;
144  char32_t lookahead = '\0';
145  for (char32_t c : uinput) {
146  if (lookahead != '\0') {
147  // if we have a lookahead, check the htmlLookahead map if there is an entity for that
148  // (there should always be one, if the map is correct)
149  if (htmlLookaheads.count(lookahead) != 1) {
150  output << cv.to_bytes(lookahead);
151  lookahead = '\0';
152  } else {
153  // check whether the current char matches the second char for the entity
154  auto& currentLA = htmlLookaheads.at(lookahead);
155  if (currentLA.first == c) {
156  output << cv.to_bytes(currentLA.second);
157  lookahead = '\0';
158  continue;
159  } else {
160  output << cv.to_bytes(lookahead);
161  lookahead = '\0';
162  }
163  }
164  }
165  if (htmlEntities.count(c) == 1) {
166  if (htmlEntities.at(c).empty()) {
167  // empty entity string means that entity stands for 2 characters
168  lookahead = c;
169  } else {
170  output << cv.to_bytes(htmlEntities.at(c));
171  }
172  } else {
173  output << cv.to_bytes(c);
174  }
175  }
176  // if lookahead is not \0, the last char has been (falsely) identified as lookahead
177  if (lookahead != '\0') {
178  output << cv.to_bytes(lookahead);
179  }
180  input = output.str();
181  }
182  return input;
